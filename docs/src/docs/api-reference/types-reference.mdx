## Type Reference

Complete TypeScript type definitions for Orbis plugin development.

## Core Types

### PageDefinition

Complete page schema definition.

<CodeBlock lang="typescript">
```typescript
interface PageDefinition {
  id: string;
  title: string;
  description?: string;
  icon?: string;
  initialState?: Record<string, any>;
  stateConfig?: StateConfig;
  on_mount?: ActionSchema[];
  onUnmount?: ActionSchema[];
  layout?: LayoutConfig;
  children: ComponentSchema[];
}
```
</CodeBlock>

## Usage Examples

### Complete Page

<CodeBlock lang="typescript">
```typescript
const page: PageDefinition = {
  id: 'users',
  title: 'Users',
  initialState: {
    users: [],
    loading: false
  },
  on_mount: [
    { type: 'call_api', api: 'getUsers', storeAs: 'users' }
  ],
  children: [
    {
      type: 'PageHeader',
      title: 'User Management',
      actions: [
        {
          type: 'Button',
          text: 'Add User',
          events: {
            onClick: [{ type: 'navigate', route: '/users/new' }]
          }
        }
      ]
    },
    {
      type: 'Table',
      data: '{{state.users}}',
      columns: [
        { key: 'name', header: 'Name' },
        { key: 'email', header: 'Email' }
      ]
    }
  ]
};
```
</CodeBlock>

### Type Guards

<CodeBlock lang="typescript">
```typescript
function isButtonSchema(schema: ComponentSchema): schema is ButtonSchema {
  return schema.type === 'Button';
}

function isFormSchema(schema: ComponentSchema): schema is FormSchema {
  return schema.type === 'Form';
}
```
</CodeBlock>
